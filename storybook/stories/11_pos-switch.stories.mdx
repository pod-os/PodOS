## pos-switch

Selects a template to render based on properties of the subject uri

<Canvas
  withSource="open">
  <Story
    name="pos-switch"
  >
    {({uri}) => html`
    <pos-resource uri="${uri}">
      <pos-switch>
        <pos-case if-typeof="http://www.w3.org/ns/ldp#Container"><template>Container</template></pos-case>
        <pos-case if-typeof="http://www.w3.org/ns/ldp#BasicContainer"><template>Basic Container</template></pos-case>
        <pos-case else if-typeof="http://www.w3.org/ns/ldp#Resource"><template>Resource and not a container</template></pos-case>
        <pos-case else><template>Not a resource or container</template>
      </pos-switch>
</pos-resource>
    `}
  </Story>
</Canvas>
